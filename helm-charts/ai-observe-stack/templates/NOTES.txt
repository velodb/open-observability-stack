{{- if .Values.otel.enabled }}

2. Send telemetry data to OTel Collector:

   # From within the cluster
   OTLP gRPC: {{ include "aiobs.otel.fullname" . }}.{{ .Release.Namespace }}.svc.cluster.local:{{ .Values.otel.ports.otlpGrpc }}
   OTLP HTTP: {{ include "aiobs.otel.fullname" . }}.{{ .Release.Namespace }}.svc.cluster.local:{{ .Values.otel.ports.otlpHttp }}

   # Port forward for local testing
   kubectl port-forward svc/{{ include "aiobs.otel.fullname" . }} 4317:{{ .Values.otel.ports.otlpGrpc }} 4318:{{ .Values.otel.ports.otlpHttp }} -n {{ .Release.Namespace }}
{{- end }}

{{- if eq .Values.doris.mode "internal" }}

3. Access Doris:

   # Port forward to Doris FE (MySQL protocol)
   kubectl port-forward svc/{{ include "aiobs.doris.clusterName" . }}-fe 9030:9030 -n {{ .Release.Namespace }}

   # Connect via MySQL client
   mysql -h 127.0.0.1 -P 9030 -u root

   # Port forward to Doris FE Web UI
   kubectl port-forward svc/{{ include "aiobs.doris.clusterName" . }}-fe 8030:8030 -n {{ .Release.Namespace }}
   # Open http://localhost:8030
{{- end }}
